Assuming “mali” means **Malli**, here’s a directly usable draft: Malli schemas for the pipeline entities, a compact set of μ executable contracts, and a state-machine spec that fits your two-loop design. In your role model, the mode-specific primary agent maps cleanly to a **Transcriber** in scribe mode and a **Producer** in composition mode, while **copyist** is adjacent but narrower because copyists usually prepare readable parts from existing notation rather than deriving them from recordings. [online.berklee](https://online.berklee.edu/careers-in-music/roles/transcriber)

## Names

That role split is defensible in music-industry terms: planning/direction, transcription or production execution, adjudication, QC review, and final outside audit all map to recognizable production functions. I’d keep one superclass, `:primary-agent`, and specialize it by mode as `:transcriber` or `:producer`. [asmac](https://asmac.org/about/what-we-do/)

## Malli schemas

This version is normalized around handoff packets, claims, reviews, and transitions. It is intentionally boring: explicit names, small maps, and enough structure to support replay, audit, and restart.

```clojure
(ns music.pipeline.schema
  (:require
    [malli.core :as m]))

(def Role
  [:enum
   :planner-agent
   :primary-agent
   :transcriber
   :producer
   :adjudication-agent
   :qc-reviewer-agent
   :human-auditor])

(def Mode
  [:enum :scribe :composition])

(def Severity
  [:enum :info :minor :major :blocking])

(def Verdict
  [:enum :accept :revise :reject])

(def Domain
  [:enum
   :lyrics-realized
   :word-syllable-timing
   :pitch-content
   :ornament-articulation
   :structure
   :render-fidelity
   :instruction-following
   :prompt-audio-discrepancy])

(def ArtifactKind
  [:enum
   :audio
   :stem
   :lyrics
   :phonemes
   :note-events
   :midi
   :musicxml
   :ustx
   :spectrogram
   :report
   :reference-catalog])

(def Span
  [:map
   [:t0 {:optional true} number?]
   [:t1 {:optional true} number?]
   [:bars {:optional true} [:vector pos-int?]]
   [:beats {:optional true} [:vector number?]]
   [:track {:optional true} keyword?]])

(def ArtifactRef
  [:map
   [:artifact/id string?]
   [:artifact/kind ArtifactKind]
   [:artifact/path string?]
   [:artifact/label {:optional true} string?]
   [:artifact/source-job-id {:optional true} string?]
   [:artifact/span {:optional true} Span]
   [:artifact/metadata {:optional true} [:map-of keyword? any?]]])

(def Ambiguity
  [:map
   [:ambiguity/id string?]
   [:ambiguity/type keyword?]
   [:ambiguity/domain [:vector Domain]]
   [:ambiguity/severity [:double {:min 0.0 :max 1.0}]]
   [:ambiguity/note string?]
   [:ambiguity/policy
    [:map
     [:reduce-penalty {:optional true} [:double {:min 0.0 :max 1.0}]]
     [:require-human-audit-if-disagreement-gt {:optional true}
      [:double {:min 0.0 :max 1.0}]]
     [:suppress-domain {:optional true} [:vector Domain]]]]])

(def Task
  [:map
   [:task/id string?]
   [:task/mode Mode]
   [:task/role Role]
   [:task/objective string?]
   [:task/domain {:optional true} Domain]
   [:task/span {:optional true} Span]
   [:task/inputs [:vector ArtifactRef]]
   [:task/constraints {:optional true} [:map-of keyword? any?]]
   [:task/success-criteria {:optional true} [:vector string?]]
   [:task/ambiguities {:optional true} [:vector Ambiguity]]])

(def Plan
  [:map
   [:plan/id string?]
   [:job/id string?]
   [:plan/version pos-int?]
   [:plan/mode Mode]
   [:plan/created-by [:enum :planner-agent]]
   [:plan/objective string?]
   [:plan/tasks [:vector Task]]
   [:plan/global-ambiguities {:optional true} [:vector Ambiguity]]
   [:plan/restart-context {:optional true}
    [:map
     [:previous-plan-id {:optional true} string?]
     [:failed-artifacts {:optional true} [:vector ArtifactRef]]
     [:latest-review-id {:optional true} string?]
     [:latest-audit-id {:optional true} string?]]]])

(def Observation
  [:map
   [:observation/tool keyword?]
   [:observation/raw any?]
   [:observation/confidence {:optional true} [:double {:min 0.0 :max 1.0}]]
   [:observation/difficulty {:optional true} [:double {:min 0.0 :max 1.0}]]
   [:observation/rationale {:optional true} string?]
   [:observation/provenance {:optional true} [:vector ArtifactRef]]])

(def Claim
  [:map
   [:claim/id string?]
   [:claim/domain Domain]
   [:claim/target keyword?]
   [:claim/span Span]
   [:claim/hypothesis string?]
   [:claim/observations [:vector Observation]]
   [:claim/score {:optional true} [:double {:min 0.0 :max 1.0}]]
   [:claim/disagreement {:optional true} [:double {:min 0.0 :max 1.0}]]
   [:claim/ambiguity-impact {:optional true} [:double {:min 0.0 :max 1.0}]]
   [:claim/status [:enum :pending :scored :contested :accepted :rejected]]])

(def ArtifactBundle
  [:map
   [:bundle/id string?]
   [:job/id string?]
   [:bundle/mode Mode]
   [:bundle/produced-by Role]
   [:bundle/artifacts [:vector ArtifactRef]]
   [:bundle/open-issues {:optional true} [:vector string?]]
   [:bundle/coverage {:optional true}
    [:map
     [:bars-total {:optional true} pos-int?]
     [:bars-covered {:optional true} pos-int?]
     [:claims-total {:optional true} pos-int?]
     [:claims-covered {:optional true} pos-int?]]]])

(def ReviewNote
  [:map
   [:note/id string?]
   [:note/domain Domain]
   [:note/severity Severity]
   [:note/span {:optional true} Span]
   [:note/issue string?]
   [:note/required-action string?]
   [:note/preferred-hypothesis {:optional true} string?]])

(def Review
  [:map
   [:review/id string?]
   [:job/id string?]
   [:review/by [:enum :qc-reviewer-agent :human-auditor]]
   [:review/verdict Verdict]
   [:review/severity Severity]
   [:review/notes [:vector ReviewNote]]
   [:review/restart-from {:optional true}
    [:enum :planner :primary :adjudication :qc-review :human-audit]]])

(def ReferenceEntry
  [:map
   [:reference/id string?]
   [:reference/source-audio ArtifactRef]
   [:reference/accepted-bundle ArtifactBundle]
   [:reference/reviews [:vector Review]]
   [:reference/tags {:optional true} [:vector string?]]
   [:reference/languages {:optional true} [:vector keyword?]]
   [:reference/usable-for [:vector Mode]]])

(def ReferenceCatalog
  [:map
   [:catalog/id string?]
   [:catalog/version pos-int?]
   [:catalog/entries [:vector ReferenceEntry]]])

(def EventType
  [:enum
   :job/created
   :plan/generated
   :task/dispatched
   :artifacts/submitted
   :claims/scored
   :qc/accepted
   :qc/revise-requested
   :references/cataloged
   :composition/started
   :human-audit/accepted
   :human-audit/rejected
   :job/completed
   :job/failed])

(def Event
  [:map
   [:event/id string?]
   [:job/id string?]
   [:event/type EventType]
   [:event/at inst?]
   [:event/by Role]
   [:event/payload [:map-of keyword? any?]]])

(def Job
  [:map
   [:job/id string?]
   [:job/mode Mode]
   [:job/state keyword?]
   [:job/objective string?]
   [:job/inputs [:vector ArtifactRef]]
   [:job/plan {:optional true} Plan]
   [:job/current-bundle {:optional true} ArtifactBundle]
   [:job/current-claims {:optional true} [:vector Claim]]
   [:job/reviews {:optional true} [:vector Review]]
   [:job/reference-catalog {:optional true} ReferenceCatalog]
   [:job/events {:optional true} [:vector Event]]])
```

## μ contracts

These are the epistemic contracts: small, testable, restart-safe. I’d treat them as preconditions on transitions plus acceptance criteria on artifacts.

```clojure
(def MuSpecs
  [{:mu/id :mu-1
    :title "Accepted bundle must be auditable"
    :rule  "Any accepted bundle includes provenance, source span coverage, and open-issues list."}

   {:mu/id :mu-2
    :title "Revision must be actionable"
    :rule  "Any :revise verdict contains at least one note with :note/required-action."}

   {:mu/id :mu-3
    :title "Composition uses approved references only"
    :rule  "Composition mode may only consume entries present in the accepted reference catalog."}

   {:mu/id :mu-4
    :title "Contested claims escalate"
    :rule  "If claim disagreement exceeds threshold and ambiguity severity is high, route to human audit."}

   {:mu/id :mu-5
    :title "Primary agent is mode-specialized"
    :rule  "In :scribe mode, primary agent role resolves to :transcriber; in :composition mode, to :producer."}

   {:mu/id :mu-6
    :title "Prompt text does not override sung realization"
    :rule  "In scribe mode, lyrics-realized scoring prefers performed audio over original prompt text when they diverge."}

   {:mu/id :mu-7
    :title "Rejected outputs preserve restart context"
    :rule  "Any restart carries prior plan, failed artifacts, latest review, and latest audit feedback."}

   {:mu/id :mu-8
    :title "No silent acceptance"
    :rule  "A bundle cannot transition to accepted without adjudication output and review verdict."}

   {:mu/id :mu-9
    :title "Human rejection is structured"
    :rule  "Human auditor rejection must identify domain, span or section, severity, and issue text."}

   {:mu/id :mu-10
    :title "Reference catalog is append-only by version"
    :rule  "Accepted references are versioned; supersession creates a new catalog version rather than mutating history."}])
```

If you want these as executable validators:

```clojure
(defn accepted-bundle-auditable? [{:bundle/keys [artifacts coverage open-issues]}]
  (and (seq artifacts)
       (map? coverage)
       (vector? open-issues)))

(defn revise-actionable? [{:review/keys [verdict notes]}]
  (if (= verdict :revise)
    (some #(seq (:note/required-action %)) notes)
    true))

(defn composition-approved-refs-only? [{:job/keys [mode reference-catalog inputs]}]
  (if (= mode :composition)
    (let [approved-ids (->> (:catalog/entries reference-catalog)
                            (map :reference/id)
                            set)]
      (every? #(or (not= (:artifact/kind %) :reference-catalog)
                   (contains? approved-ids (:artifact/id %)))
              inputs))
    true))
```

## State machine

This machine keeps both loops in one graph and makes restart behavior explicit. The important design choice is that `:qc-review` and `:human-audit` do not mutate artifacts; they only emit verdicts that either advance or restart.

```clojure
(def State
  [:enum
   :job-created
   :planning
   :scribe-executing
   :scribe-adjudicating
   :scribe-qc-review
   :reference-cataloged
   :composition-executing
   :composition-adjudicating
   :composition-qc-review
   :human-audit
   :accepted
   :rejected
   :failed])

(def MachineEvent
  [:enum
   :begin-plan
   :plan-ready
   :submit-artifacts
   :score-claims
   :qc-accept
   :qc-revise
   :catalog-reference
   :start-composition
   :audit-accept
   :audit-reject
   :fail])

(def StateMachine
  [:map
   [:machine/id keyword?]
   [:machine/initial State]
   [:machine/states [:vector State]]
   [:machine/events [:vector MachineEvent]]
   [:machine/transitions
    [:vector
     [:map
      [:from State]
      [:event MachineEvent]
      [:to State]
      [:guard {:optional true} keyword?]
      [:action {:optional true} keyword?]]]]
   [:machine/guards {:optional true} [:map-of keyword? fn?]]
   [:machine/actions {:optional true} [:map-of keyword? fn?]]])

(def pipeline-machine
  {:machine/id :music-pipeline
   :machine/initial :job-created
   :machine/states
   [:job-created
    :planning

    :reference-cataloged
    :composition-executing
    :composition-adjudicating
    :composition-qc-review
    :human-audit
    :accepted
    :rejected
    :failed]

   :machine/events
   [:begin-plan
    :plan-ready
    :submit-artifacts
    :score-claims
    :qc-accept
    :qc-revise
    :catalog-reference
    :start-composition
    :audit-accept
    :audit-reject
    :fail]

   :machine/transitions
   [{:from :job-created
     :event :begin-plan
     :to :planning}

    {:from :planning
     :event :plan-ready
     :to :scribe-executing
     :guard :scribe-mode?}

    {:from :scribe-executing
     :event :submit-artifacts
     :to :scribe-adjudicating
     :guard :bundle-valid?}

    {:from :scribe-adjudicating
     :event :score-claims
     :to :scribe-qc-review
     :guard :claims-valid?}

    {:from :scribe-qc-review
     :event :qc-revise
     :to :planning
     :guard :review-actionable?}

    {:from :scribe-qc-review
     :event :qc-accept
     :to :reference-cataloged
     :guard :accepted-bundle-auditable?}

    {:from :reference-cataloged
     :event :start-composition
     :to :composition-executing
     :guard :has-approved-references?}

    {:from :planning
     :event :plan-ready
     :to :composition-executing
     :guard :composition-mode?}

    {:from :composition-executing
     :event :submit-artifacts
     :to :composition-adjudicating
     :guard :bundle-valid?}

    {:from :composition-adjudicating
     :event :score-claims
     :to :composition-qc-review
     :guard :claims-valid?}

    {:from :composition-qc-review
     :event :qc-revise
     :to :planning
     :guard :review-actionable?}

    {:from :composition-qc-review
     :event :qc-accept
     :to :human-audit}

    {:from :human-audit
     :event :audit-reject
     :to :planning
     :guard :human-rejection-structured?}

    {:from :human-audit
     :event :audit-accept
     :to :accepted}

    {:from :job-created
     :event :fail
     :to :failed}

    {:from :planning
     :event :fail
     :to :failed}

    {:from :scribe-executing
     :event :fail
     :to :failed}

    {:from :scribe-adjudicating
     :event :fail
     :to :failed}

    {:from :scribe-qc-review
     :event :fail
     :to :failed}

    {:from :reference-cataloged
     :event :fail
     :to :failed}

    {:from :composition-executing
     :event :fail
     :to :failed}

    {:from :composition-adjudicating
     :event :fail
     :to :failed}

    {:from :composition-qc-review
     :event :fail
     :to :failed}

    {:from :human-audit
     :event :fail
     :to :failed}]})
```

## Guards

These should stay tiny and boring. A state machine becomes readable when the complexity lives in data and guards, not in giant transition actions.

```clojure
(defn scribe-mode? [job]
  (= (:job/mode job) :scribe))

(defn composition-mode? [job]
  (= (:job/mode job) :composition))

(defn bundle-valid? [job]
  (m/validate ArtifactBundle (:job/current-bundle job)))

(defn claims-valid? [job]
  (every? #(m/validate Claim %) (:job/current-claims job)))

(defn review-actionable? [job]
  (when-let [review (last (:job/reviews job))]
    (revise-actionable? review)))

(defn has-approved-references? [job]
  (seq (get-in job [:job/reference-catalog :catalog/entries])))

(defn human-rejection-structured? [job]
  (when-let [review (last (:job/reviews job))]
    (and (= (:review/by review) :human-auditor)
         (= (:review/verdict review) :reject)
         (seq (:review/notes review)))))
```
